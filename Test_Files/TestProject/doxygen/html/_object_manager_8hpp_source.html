241 template<typename C, typename ... Args>
242 ID ObjectManager::addComponents(ID objectID, const C& component, Args&& ... others)
243 {
244  registerComponent<C>();
245 
246  //A counter for the total components added to the object
247  ID added = 0;
248 
249  if(objectID < getTotalObjects())
250  {
251  //Only add the component if the object does not have an instance of it already.
252  if(objects[objectID].componentIndices.find(C::getFamily()) == objects[objectID].componentIndices.end())
253  {
254  //Add the component to its array
255 
256  ID componentIndex = getComponentArray<C>().add_item(component);
257  getComponentArray<C>()[componentIndex].ownerID = objectID;
258 
259  //Store a pointer to the components PackedArray for destroying of the object
260  objects[objectID].componentArrays[C::getFamily()] = &getComponentArray<C>();
261 
262  //Store the component's index in the object
263  objects[objectID].componentIndices[C::getFamily()] = componentIndex;
264 
265  added = 1;
266  }
267  }
268 
269  //If there are more objects to add then add them.
270  if(sizeof ... (others) > 0)
271  return added + addComponents(objectID, others...);
272 
273  return added;
274 }

319 template<typename C, typename ... Args>
320 ID ObjectManager::removeComponents(ID objectID)
321 {
322  ID totalRemoved = 0;
323  if(!endRecursion(C()))
324  {
325  if(objectID < getTotalObjects())
326  {
327  //Only remove the component if the object has an instance of it.
328  if(objects[objectID].componentIndices.find(C::getFamily()) != objects[objectID].componentIndices.end())
329  {
330  ID componentIndex = objects[objectID].componentIndices[C::getFamily()];
331 
332  //Remove the component from its array
333  getComponentArray<C>().remove(componentIndex);
334 
335  //Remove the pointer to the component maps from the object
336  objects[objectID].componentArrays.erase(C::getFamily());
337  objects[objectID].componentIndices.erase(C::getFamily());
338 
339  totalRemoved = 1;
340  }
341 
342  return totalRemoved + removeComponents<Args...>(objectID);
343  }
344  }
345 
346  return totalRemoved;
347 }

448 template<typename C, typename ... Args>
449 void ObjectManager::addComponentsToPrototype(const std::string& prototypeName, const C& first, Args&& ... others)
450 {
451  registerComponent<C>();
452 
453  //Get the prototype
454  auto& prototype = objectPrototypes[prototypeName];
455 
456  if(prototype.objectID == ID(-1))
457  prototype.objectID = prototypeIDCounter++;
458 
459  //If the prototype does not already have the given component
460  if(prototype.componentArrays.find(C::getFamily()) == prototype.componentArrays.end())
461  {
462  //Add the first component in the list
463  auto compIdx = getPrototypeComponentArray<C>().add_item(first);
464 
465  prototype.componentArrays[C::getFamily()] = &getPrototypeComponentArray<C>();
466  prototype.componentIndices[C::getFamily()] = compIdx;
467 
468  //If there are more components to add, keep adding them
469  if(sizeof...(others) > 0)
470  addComponentsToPrototype(prototypeName, others...);
471  }
472 
473 }
